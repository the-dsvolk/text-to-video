1. Get the application URL by running these commands:
{{- if .Values.fastapi.ingress.enabled }}
{{- range $host := .Values.fastapi.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.fastapi.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.fastapi.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "text-to-video.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "text-to-video.fullname" . }}-fastapi)
  export NODE_IP=$(kubectl get nodes --namespace {{ include "text-to-video.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.fastapi.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ include "text-to-video.namespace" . }} svc -w {{ include "text-to-video.fullname" . }}-fastapi'
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "text-to-video.namespace" . }} {{ include "text-to-video.fullname" . }}-fastapi --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.fastapi.service.port }}
{{- else if contains "ClusterIP" .Values.fastapi.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "text-to-video.namespace" . }} -l "{{ include "text-to-video.fastapi.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "text-to-video.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ include "text-to-video.namespace" . }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}

2. Test the API:

   # Submit a video generation job
   curl -X POST "http://<YOUR_SERVICE_IP>/generate-video/" \
     -H "Content-Type: application/json" \
     -d '{"prompt": "A robot painting a masterpiece, cinematic style"}'

   # Check job status (replace JOB_ID with the returned job_id)
   curl "http://<YOUR_SERVICE_IP>/status/<JOB_ID>"

   # Download the video when complete
   curl -o video.mp4 "http://<YOUR_SERVICE_IP>/download/<JOB_ID>"

3. View the API documentation:
   Visit http://<YOUR_SERVICE_IP>/docs for the interactive API documentation.

4. Monitor the services:
   kubectl get pods -n {{ include "text-to-video.namespace" . }}
   kubectl logs -f -n {{ include "text-to-video.namespace" . }} deployment/{{ include "text-to-video.fullname" . }}-fastapi
